Chaining Multiple Decorators
Sun 25 January 2026 
def bold(func):
    def wrapper():
        print("<b>")
        func()
        print("</b>")
    return wrapper

def italic(func):
    def wrapper():
        print("<i>")
        func()
        print("</i>")
    return wrapper

@bold
@italic
def display():
    print("Text")

display()
