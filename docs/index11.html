Cooperative Multiple Inheritance Using Super()
Sun 25 January 2026 
class A:
    def __init__(self):
        print("A init")
        super().__init__()

class B:
    def __init__(self):
        print("B init")
        super().__init__()

class C(A, B):
    def __init__(self):
        print("C init")
        super().__init__()

c = C()

C init
A init
B init
