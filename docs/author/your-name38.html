Real-World Decorator Example (Authentication)
Sun 25 January 2026 
def require_login(func):
    def wrapper(user):
        if not user.get("authenticated"):
            return "Access Denied"
        return func(user)
    return wrapper

@require_login
def view_profile(user):
    return f"Welcome {user['name']}"

user = {"name": "Alice", "authenticated": True}
print(view_profile(user))

Welcome Alice

def read_large_file(filename):
    with open(filename, "r") as file:
        for line in file:
            yield line.strip()

for line in read_large_file("data.csv"):
    print(line)

Real-World Iterator Example
Sun 25 January 2026 
class FileLineReader:
    def __init__(self, filename):
        self.file = open(filename, "r")

    def __iter__(self):
        return self

    def __next__(self):
        line = self.file.readline()
        if line:
            return line.strip()
        self.file.close()
        raise StopIteration

for line in FileLineReader("data.csv"):
    print(line)

import re

def validate_phone(phone):
    pattern = r"^\+?\d{10,13}$"
    return bool(re.match(pattern, phone))

print(validate_phone("+911234567890"))  # True
print(validate_phone("12345"))          # False
