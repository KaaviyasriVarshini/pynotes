Real-World Polymorphism Example
Sun 25 January 2026 
class Notification:
    def send(self):
        print("Sending notification")

class Email(Notification):
    def send(self):
        print("Sending Email")

class SMS(Notification):
    def send(self):
        print("Sending SMS")

def notify(service):
    service.send()

notify(Email())
notify(SMS())

Sending Email
Sending SMS

Real-World Regex Example (Phone Validation)
Sun 25 January 2026 
import re

def validate_phone(phone):
    pattern = r"^\+?\d{10,13}$"
    return bool(re.match(pattern, phone))

print(validate_phone("+911234567890"))  # True
print(validate_phone("12345"))          # False

    result = 1
    for i in range(1, n + 1):
        result *= i
    return result

Recursive Fibonacci Series
Sun 25 January 2026 
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(6))  # Output: 8

def reverse_string(text):
    if len(text) == 0:
        return text
    return reverse_string(text[1:]) + text[0]

print(reverse_string("Python"))  # Output: nohtyP
