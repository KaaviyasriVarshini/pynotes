Enterprise Use Case: Rate Limiting Logic
Sun 25 January 2026 
import time

last_request = time.time()

def can_execute():
    global last_request
    current = time.time()
    if current - last_request >= 5:
        last_request = current
        return True
    return False

print(can_execute())
