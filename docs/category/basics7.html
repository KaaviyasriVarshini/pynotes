Class-Based Decorator
Sun 25 January 2026 
class Logger:
    def __init__(self, func):
        self.func = func

    def __call__(self, *args, **kwargs):
        print("Calling function...")
        return self.func(*args, **kwargs)

@Logger
def show():
    print("Processing data")

show()

Closure Retaining State
Sun 25 January 2026 
def counter():
    count = 0

    def increment():
        nonlocal count
        count += 1
        return count

    return increment

c = counter()
print(c())  # 1
print(c())  # 2
