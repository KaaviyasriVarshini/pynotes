Nonlocal In Closures

def power(base):
    exponent = 2

    def calculate():
        nonlocal exponent
        exponent += 1
        return base ** exponent

    return calculate

p = power(2)
print(p())
print(p())

Optimizing Recursion With Memoization
Sun 25 January 2026 
from functools import lru_cache

@lru_cache(maxsize=None)
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(30))  # Optimized recursion

832040
