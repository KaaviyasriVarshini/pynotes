Decorator For Timing Execution

Sun 25 January 2026 
import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print("Execution Time:", time.time() - start)
        return result
    return wrapper

@timer
def process():
    for _ in range(1000000):
        pass

process()
